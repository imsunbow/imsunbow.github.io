전표 밸런스 검증 (ZE1FI0001)

임시전표 저장 전 차변/대변 합계가 일치하는지 검증하는 로직

" 전표 아이템 내역에서 차변/대변 합계 계산
DATA: lv_debit  TYPE bseg-dmbtr,
      lv_credit TYPE bseg-dmbtr.

LOOP AT gt_item INTO gs_item.
  CASE gs_item-shkzg.
    WHEN 'S'.  " 차변 (Soll)
      lv_debit = lv_debit + gs_item-dmbtr.
    WHEN 'H'.  " 대변 (Haben)
      lv_credit = lv_credit + gs_item-dmbtr.
  ENDCASE.
ENDLOOP.

" 차대변 불일치 시 저장 차단
IF lv_debit NE lv_credit.
  MESSAGE '차변과 대변 합계가 일치하지 않습니다.'
    TYPE 'E' DISPLAY LIKE 'E'.
  RETURN.
ENDIF.

" 밸런스 통과 → 번호 범위 채번 후 DB 저장
CALL FUNCTION 'NUMBER_GET_NEXT'
  EXPORTING
    nr_range_nr = '01'
    object      = 'ZFI_TEMP'
  IMPORTING
    number      = lv_docno.

gs_header-belnr = lv_docno.
INSERT zfi_temp_h FROM gs_header.
INSERT zfi_temp_i FROM TABLE gt_item.

결재 상태 처리 (ZRE1FI0008)

임시전표 승인/반려 시 상태값을 업데이트하고 정식 전기로 연계하는 로직

" 결재 액션에 따라 전표 상태 분기
CASE lv_action.
  WHEN 'APPROVE'.
    " 승인: 상태를 P(완료)로 변경 후 정식 전기 연계
    UPDATE zfi_temp_h
      SET   status = 'P'
            approved_by   = sy-uname
            approved_date = sy-datum
      WHERE belnr = lv_belnr.

    " BAPI로 FI 정식 전표 생성
    CALL FUNCTION 'BAPI_ACC_DOCUMENT_POST'
      EXPORTING
        documentheader = ls_header
      TABLES
        accountgl      = lt_accountgl
        currencyamount = lt_curramt
        return         = lt_return.

    CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'
      EXPORTING wait = 'X'.

  WHEN 'REJECT'.
    " 반려: 상태를 J로 변경, 반려 사유 저장
    UPDATE zfi_temp_h
      SET   status      = 'J'
            reject_note = lv_reason
            reject_date = sy-datum
      WHERE belnr = lv_belnr.
ENDCASE.

감가상각 계산 (ZE1FI0002)

고정자산 현황에서 취득원가·상각률로 장부가액을 실시간 계산하는 로직

" 자산별 감가상각 누계액 및 장부가액 계산
LOOP AT gt_asset INTO gs_asset.

  " 경과 연수 = 현재 회계연도 - 취득 연도
  lv_years = sy-datum+0(4) - gs_asset-acquisition_date+0(4).

  " 정액법 상각: 취득원가 / 내용 연수
  IF gs_asset-useful_life GT 0.
    lv_annual_dep = gs_asset-acquisition_cost
                    / gs_asset-useful_life.

    lv_accum_dep  = lv_annual_dep * lv_years.

    " 누계액이 취득원가를 초과하지 않도록 제한
    IF lv_accum_dep GT gs_asset-acquisition_cost.
      lv_accum_dep = gs_asset-acquisition_cost.
    ENDIF.
  ENDIF.

  " 장부가액 = 취득원가 - 감가상각 누계액
  gs_asset-book_value = gs_asset-acquisition_cost
                        - lv_accum_dep.
  gs_asset-accum_dep  = lv_accum_dep.

  MODIFY gt_asset FROM gs_asset.
ENDLOOP.

OData 상태별 집계 (zgwe1fi0001 · OverviewView)

백엔드 OData에서 전체 데이터를 읽어 상태별 카운트를 JSON 모델에 반영

_calculateChartCounts: function (oModel) {
  var oChartModel = this.getView().getModel("chart");

  // 리스트 필터에 무관하게 원본 데이터 직접 조회
  oModel.read("/ApprovalHeaderSet", {
    success: function (oData) {
      var iWait = 0, iDone = 0, iReject = 0;

      oData.results.forEach(function (item) {
        if      (item.Status === "D") { iWait++;   }
        else if (item.Status === "P") { iDone++;   }
        else if (item.Status === "J") { iReject++; }
      });

      // 집계 결과 → JSON 모델 반영 → XML 차트 자동 갱신
      oChartModel.setProperty("/WaitCount",   iWait);
      oChartModel.setProperty("/DoneCount",   iDone);
      oChartModel.setProperty("/RejectCount", iReject);
    }.bind(this),
    error: function () {
      sap.m.MessageToast.show("집계 데이터 로드 실패");
    }
  });
},

클라이언트 사이드 필터링 (zgwe1fi0002)

OData 서버 에러 회피를 위해 전체 데이터를 로드 후 클라이언트에서 필터링

onOverviewSearch: function () {
  var sPartner = this.byId("ovInputPartner").getValue().trim();
  var sBukrs   = this.byId("ovInputBukrs").getValue().trim();
  var aFilters = [];

  if (sPartner) {
    aFilters.push(new Filter("PartnerNo", FilterOperator.EQ, sPartner));
  }
  if (sBukrs) {
    aFilters.push(new Filter("Bukrs", FilterOperator.EQ, sBukrs));
  }

  // 서버 필터 대신 클라이언트 필터로 집계 재계산
  this._loadDunningStats(aFilters);
},

onOverviewReset: function () {
  this.byId("ovInputPartner").setValue("");
  this.byId("ovInputBukrs").setValue("");
  this._loadDunningStats([]);
},
